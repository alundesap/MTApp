VIEW "VW_DATASET_FIELD_UDV" ( "DATASET_FIELD_UDV_ID",
	 "DATASET_ID",
	 "DATASET_FIELD_VALIDATION_ID",
	 "DATASET_FIELD_ID",
	 "DM_DATASET_TYPE_ID",
	 "DM_NK_FIELD_ID",
	 "UDV",
	 "OBLIGATION_ID",
	 "TAX_PERIOD_ID",
	 "EXCEPTION_CODE",
	 "IS_APPROVED",
	 "COMMENT",
	 "ROW_NUM" ) AS SELECT
	 DATASET_FIELD_UDV."DATASET_FIELD_UDV_ID",
	 DATASET_FIELD_UDV."DATASET_ID",
	 DATASET_FIELD_VALIDATION."DATASET_FIELD_VALIDATION_ID",
	 DATASET_FIELD_VALIDATION."DATASET_FIELD_ID",
	 DATASET_FIELD_VALIDATION."DM_DATASET_TYPE_ID" ,
	 DATASET_FIELD_VALIDATION."DM_NK_FIELD_ID" ,
	 DATASET_FIELD_UDV."UDV" ,
	 DATASET_FIELD_VALIDATION."OBLIGATION_ID" ,
	 OBLIGATION."TAX_PERIOD_ID",
	 DATASET_FIELD_VALIDATION."EXCEPTION_CODE" ,
	 DATASET_FIELD_UDV."IS_APPROVED",
	 DATASET_FIELD_UDV."COMMENT",
	 ROW_NUMBER() OVER (PARTITION BY OBLIGATION."TAX_PERIOD_ID",
	 DATASET_FIELD_VALIDATION."DATASET_FIELD_ID",
	 DATASET_FIELD_VALIDATION."DM_DATASET_TYPE_ID",
	 DATASET_FIELD_VALIDATION."DM_NK_FIELD_ID" 
	ORDER BY DATASET_FIELD_UDV."DATASET_ID" DESC) AS ROW_NUM 
FROM "DATASET_FIELD_VALIDATION" DATASET_FIELD_VALIDATION 
INNER JOIN "OBLIGATION" OBLIGATION ON DATASET_FIELD_VALIDATION."OBLIGATION_ID" = OBLIGATION."OBLIGATION_ID" 
AND OBLIGATION."STATUS_ID" = 1 
INNER JOIN "DATASET_FIELD_UDV" DATASET_FIELD_UDV ON DATASET_FIELD_VALIDATION."DATASET_FIELD_VALIDATION_ID" = DATASET_FIELD_UDV."DATASET_FIELD_VALIDATION_ID" 
AND DATASET_FIELD_UDV."STATUS_ID" = 1 WITH READ ONLY
