VIEW "VW_USER_ROLE_ACTIVITY" ( "USER_ROLE_ACTIVITY_ID",
	 "USER_ROLE_ID",
	 "USER_ROLE_NAME",
	 "USER_ROLE_DESCRIPTION",
	 "USER_ACTIVITY_ID",
	 "USER_ACTIVITY_CODE",
	 "USER_ACTIVITY_NAME",
	 "USER_ACTIVITY_DESCRIPTION" ) AS SELECT
	 "USER_ROLE_ACTIVITY_ID",
	 R."USER_ROLE_ID",
	 "USER_ROLE_NAME",
	 R."DESCRIPTION" AS "USER_ROLE_DESCRIPTION",
	 A."USER_ACTIVITY_ID",
	 A."USER_ACTIVITY_CODE",
	 "USER_ACTIVITY_NAME",
	 A."DESCRIPTION" AS "USER_ACTIVITY_DESCRIPTION" 
FROM USER_ACTIVITY A 
LEFT JOIN USER_ROLE_ACTIVITY URA ON URA.USER_ACTIVITY_ID = A.USER_ACTIVITY_ID 
AND URA.STATUS_ID = 1 
LEFT JOIN USER_ROLE R ON R.USER_ROLE_ID = URA.USER_ROLE_ID 
AND R.STATUS_ID = 1 
WHERE A.STATUS_ID = 1 WITH READ ONLY
