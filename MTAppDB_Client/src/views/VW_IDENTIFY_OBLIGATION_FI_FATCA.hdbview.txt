VIEW "VW_IDENTIFY_OBLIGATION_FI_FATCA" ( "ZERO",
	 "ENTITY_ID",
	 "FI_ID",
	 "FATCA_STATUS_ID",
	 "TAX_PERIOD_ID",
	 "OBLIGATION_YEAR",
	 "IS_USWA",
	 "FI_JURISDICTION",
	 "FATCA_CH4_STATUS",
	 "IS_FORMATION_WITHIN_PERIOD" ) AS SELECT
	 0 AS "ZERO",
	 ENTITY."ENTITY_ID" AS "ENTITY_ID",
	 ENTITY."FI_ID" AS "FI_ID",
	 ENTITY."FATCA_STATUS_ID" AS "FATCA_STATUS_ID",
	 ENTITY."TAX_PERIOD_ID" AS "TAX_PERIOD_ID",
	 ENTITY."TAX_PERIOD_CODE" AS "OBLIGATION_YEAR",
	 ENTITY."IS_USWA" AS "IS_USWA",
	 MAP ( ENTITY.ENTITY_JURISDICTION."ISO_CODE2",
	 'NULL',
	 'NONE',
	 NULL,
	 'NONE',
	 ENTITY.ENTITY_JURISDICTION."ISO_CODE2" ) AS "FI_JURISDICTION",
	 MAP ( ENTITY.ENTITY_FATCA_STATUS."FATCA_STATUS_CODE",
	 'NULL',
	 'NONE',
	 NULL,
	 'NONE',
	 ENTITY.ENTITY_FATCA_STATUS."FATCA_STATUS_CODE" ) AS "FATCA_CH4_STATUS",
	 CASE WHEN ENTITY."TAX_PERIOD_END" >= ENTITY."FORMATION_DATE" 
THEN 1 
ELSE 0 
END AS "IS_FORMATION_WITHIN_PERIOD" 
FROM "VW_ENTITY_TD" AS ENTITY WITH READ ONLY
