VIEW "VW_CLIENT_USER" ( "CLIENT_USER_ID",
	 "USER_ID",
	 "LAST_NAME",
	 "FIRST_NAME",
	 "PHONE",
	 "EMAIL",
	 "IS_EXTERNAL",
	 "IS_TERMS_OF_USE_ACCEPTED",
	 "CLIENT_ID",
	 "CLIENT_NAME",
	 "CLIENT_SCHEMA",
	 "IS_TEST_CLIENT",
	 "CURRENT_TAX_PERIOD_ID",
	 "WORKFLOW_DEFINITION_CODE",
	 "USER_ROLE_ID",
	 "USER_ROLE_NAME",
	 "USER_ROLE_DESCRIPTION",
	 "IS_ADMIN" ) AS SELECT
	 CU.CLIENT_USER_ID,
	 U.USER_ID,
	 LAST_NAME,
	 FIRST_NAME,
	 PHONE,
	 EMAIL,
	 IS_EXTERNAL,
	 IS_TERMS_OF_USE_ACCEPTED,
	 C.CLIENT_ID,
	 C.CLIENT_NAME,
	 CLIENT_SCHEMA,
	 IS_TEST_CLIENT,
	 CURRENT_TAX_PERIOD_ID,
	 WORKFLOW_DEFINITION_CODE,
	 UR.USER_ROLE_ID,
	 USER_ROLE_NAME,
	 UR.DESCRIPTION AS USER_ROLE_DESCRIPTION,
	 UR.IS_ADMIN 
FROM CLIENT_USER CU JOIN "USER" U ON U.USER_ID = CU.USER_ID 
AND U.STATUS_ID = 1 JOIN CLIENT C ON C.CLIENT_ID = CU.CLIENT_ID 
AND C.STATUS_ID = 1 JOIN USER_ROLE UR ON UR.USER_ROLE_ID = CU.USER_ROLE_ID 
AND UR.STATUS_ID = 1 
WHERE CU.STATUS_ID = 1 
ORDER BY LAST_NAME,
	 CLIENT_SCHEMA WITH READ ONLY
