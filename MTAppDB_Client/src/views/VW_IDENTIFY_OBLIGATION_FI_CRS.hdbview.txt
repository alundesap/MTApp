VIEW "VW_IDENTIFY_OBLIGATION_FI_CRS" ( "ENTITY_ID",
	 "FI_ID",
	 "TAX_PERIOD_ID",
	 "OBLIGATION_YEAR",
	 "IS_CRS_PARTICIPATING",
	 "CRS_CLASSIFICATION_CODE",
	 "IS_FORMATION_WITHIN_PERIOD" ) AS SELECT
	 ENTITY."ENTITY_ID" AS "ENTITY_ID",
	 ENTITY."FI_ID" AS "FI_ID",
	 ENTITY."TAX_PERIOD_ID" AS "TAX_PERIOD_ID",
	 ENTITY."TAX_PERIOD_CODE" AS "OBLIGATION_YEAR",
	 MAP(ENTITY.ENTITY_OBLIGATION_AGREEMENT."IS_CRS_PARTICIPATING",
	 NULL,
	 - 1,
	 1,
	 1,
	 0) AS "IS_CRS_PARTICIPATING",
	 MAP(ENTITY.ENTITY_CRS_CLASSIFICATION."CRS_CLASSIFICATION_CODE",
	 NULL,
	 'NONE',
	 'NULL',
	 'NONE',
	 ENTITY.ENTITY_CRS_CLASSIFICATION."CRS_CLASSIFICATION_CODE") AS "CRS_CLASSIFICATION_CODE",
	 CASE WHEN ENTITY."TAX_PERIOD_END" >= ENTITY."FORMATION_DATE" 
THEN 1 
ELSE 0 
END AS "IS_FORMATION_WITHIN_PERIOD" 
FROM "VW_ENTITY_TD" AS ENTITY WITH READ ONLY
