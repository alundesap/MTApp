	 "UPDATED_BY" NVARCHAR(128) DEFAULT CURRENT_USER NOT NULL ,
	 "UPDATED_ON" LONGDATE CS_LONGDATE DEFAULT CURRENT_UTCTIMESTAMP NOT NULL ,

	 "FRIENDLY_NAME" NVARCHAR(128) DEFAULT '' NOT NULL ,
PRIMARY KEY ("DATASET_FIELD_ID")) UNLOAD PRIORITY 5 AUTO MERGE
